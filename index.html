function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss   = SpreadsheetApp.getActiveSpreadsheet();
    var sh   = ss.getSheetByName(data.sheet) ||
               ss.insertSheet(data.sheet);

    if (data.action === 'replace') {
      var last = sh.getLastRow();
      if (last > 1) sh.deleteRows(2, last - 1);
      if (data.header && sh.getLastRow() === 0) {
        sh.appendRow(data.header);
      } else if (data.header && sh.getLastRow() >= 1) {
        sh.getRange(1,1,1,data.header.length).setValues([data.header]);
      }
      if (data.rows && data.rows.length > 0) {
        sh.getRange(2, 1, data.rows.length, data.rows[0].length)
          .setValues(data.rows);
      }
    } else if (data.action === 'deleteRow') {
      var vals = sh.getRange(1, 1, sh.getLastRow(), 1).getValues();
      for (var i = vals.length - 1; i >= 1; i--) {
        if (String(vals[i][0]) === String(data.entryNo)) {
          sh.deleteRow(i + 1);
          break;
        }
      }
    } else {
      if (sh.getLastRow() === 0 && data.header) {
        sh.appendRow(data.header);
      }
      sh.appendRow(data.row);
    }

    return ok_({status:'ok'});
  } catch(err) {
    return ok_({status:'error', msg: err.toString()});
  }
}

function doGet(e) {
  try {
    var ss  = SpreadsheetApp.getActiveSpreadsheet();
    var out = {status:'ok', ts: new Date().getTime(),
               products:[], stockIn:[], stockOut:[]};

    var ps = ss.getSheetByName('Products');
    if (ps && ps.getLastRow() > 1) {
      out.products = ps
        .getRange(2,1,ps.getLastRow()-1,5)
        .getValues()
        .filter(function(r){return r[0];})
        .map(function(r){
          return {code:String(r[0]),  name:String(r[1]),
                  unit:String(r[2]),  reorder:Number(r[3])||0,
                  desc:String(r[4])};
        });
    }

    var is = ss.getSheetByName('Stock IN');
    if (is && is.getLastRow() > 1) {
      out.stockIn = is
        .getRange(2,1,is.getLastRow()-1,9)
        .getValues()
        .filter(function(r){return r[0];})
        .map(function(r){
          return {no:String(r[0]),   date:String(r[1]),
                  code:String(r[2]), name:String(r[3]),
                  unit:String(r[4]), supplier:String(r[5]),
                  qty:Number(r[6]),  price:Number(r[7]),
                  total:Number(r[8])};
        });
    }

    var os = ss.getSheetByName('Stock OUT');
    if (os && os.getLastRow() > 1) {
      out.stockOut = os
        .getRange(2,1,os.getLastRow()-1,9)
        .getValues()
        .filter(function(r){return r[0];})
        .map(function(r){
          return {no:String(r[0]),   date:String(r[1]),
                  code:String(r[2]), name:String(r[3]),
                  unit:String(r[4]), issued:String(r[5]),
                  qty:Number(r[6]),  price:Number(r[7]),
                  total:Number(r[8])};
        });
    }

    return ok_(out);
  } catch(err) {
    return ok_({status:'error', msg: err.toString()});
  }
}

function ok_(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
